Task 1:

Write a function that takes your name and age as an input parameter and print it by using string concatenation.

Answer:
// console.log("Task 1")
        function test(name, age) {
            // Getting input from user
            name = prompt("Enter your name:")
            age = prompt(name + " is old?")

            // Method 1
            console.log("Hi! " + name + " your age is " + age)

            // Method 2
            let result = name.concat(" your age is ", age)
            console.log(result)
        }
